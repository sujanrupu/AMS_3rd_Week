-- Enable the pgvector extension (only needs to run once per project)
-- create extension if not exists vector;

CREATE TABLE IF NOT EXISTS escalation_kb (
    id                      BIGSERIAL PRIMARY KEY,

    -- What kind of incident this covers
    incident_category       TEXT NOT NULL,          -- e.g. "Network", "Database", "Security"
    sub_category            TEXT,                   -- e.g. "VPN", "Replication", "Ransomware"
    keywords                TEXT[],                 -- tags for fast keyword pre-filtering

    -- A short example description of this type of incident
    -- The AI's vector search runs against this field
    example_description     TEXT NOT NULL,

    -- What the org says to do for this type of incident
    recommended_team        TEXT NOT NULL,          -- e.g. "Network Ops", "DBA Team"
    recommended_level       TEXT NOT NULL,          -- L1, L2, or L3
    rationale               TEXT NOT NULL,          -- why this team + level was chosen

    -- Self-learning counters — updated automatically by human feedback loop
    human_confirmed_count   INTEGER DEFAULT 0,      -- humans agreed with AI suggestion
    human_corrected_count   INTEGER DEFAULT 0,      -- humans overrode AI suggestion
    last_human_action_at    TIMESTAMPTZ,

    -- Vector embedding of example_description (for semantic similarity search)
    -- Populated by running escalation_kb_embedding.py after inserting rows
    embedding               VECTOR(384),

    created_at              TIMESTAMPTZ DEFAULT NOW(),
    updated_at              TIMESTAMPTZ DEFAULT NOW()
);

-- Index for fast vector similarity search
CREATE INDEX IF NOT EXISTS escalation_kb_embedding_idx
    ON escalation_kb
    USING ivfflat (embedding vector_cosine_ops)
    WITH (lists = 50);

-- ============================================================
-- FUNCTION: Semantic similarity search on escalation_kb
-- Returns top K rows by cosine similarity to a given embedding
-- This is what the AI calls to search the "document"
-- ============================================================

CREATE OR REPLACE FUNCTION search_escalation_kb(
    query_embedding  VECTOR(384),
    top_k            INTEGER DEFAULT 5,
    min_similarity   FLOAT   DEFAULT 0.40
)
RETURNS TABLE (
    id                      BIGINT,
    incident_category       TEXT,
    sub_category            TEXT,
    recommended_team        TEXT,
    recommended_level       TEXT,
    rationale               TEXT,
    example_description     TEXT,
    human_confirmed_count   INTEGER,
    human_corrected_count   INTEGER,
    similarity              FLOAT
)
LANGUAGE sql STABLE AS $$
    SELECT
        id,
        incident_category,
        sub_category,
        recommended_team,
        recommended_level,
        rationale,
        example_description,
        human_confirmed_count,
        human_corrected_count,
        1 - (embedding <=> query_embedding) AS similarity
    FROM escalation_kb
    WHERE embedding IS NOT NULL
      AND 1 - (embedding <=> query_embedding) >= min_similarity
    ORDER BY embedding <=> query_embedding
    LIMIT top_k;
$$;

-- ─────────────────────────────────────────────────────────────
-- TABLE: escalation_feedback
-- Every human triage action gets recorded here.
-- This is how the AI learns over time.
-- ─────────────────────────────────────────────────────────────

CREATE TABLE IF NOT EXISTS escalation_feedback (
    id                  BIGSERIAL PRIMARY KEY,

    ticket_key          TEXT NOT NULL,              -- e.g. "AMS-123"

    -- What the AI decided
    ai_team             TEXT,                       -- NULL if quarantine (AI had no answer)
    ai_level            TEXT,                       -- L1 / L2 / L3 / NULL
    ai_confidence       TEXT,                       -- HIGH / MEDIUM / LOW

    -- What the human decided
    human_team          TEXT NOT NULL,
    human_level         TEXT NOT NULL,              -- L1 / L2 / L3
    human_reason        TEXT NOT NULL,              -- why they kept or changed the AI decision

    -- Was the AI wrong?
    was_overridden      BOOLEAN NOT NULL,           -- true if human changed AI decision
    review_type         TEXT NOT NULL,              -- 'hitl_review' | 'quarantine_review'

    created_at          TIMESTAMPTZ DEFAULT NOW()
);

-- Index for fast lookup by ticket key
CREATE INDEX IF NOT EXISTS escalation_feedback_ticket_idx
    ON escalation_feedback (ticket_key);

-- Index for learning queries (team + level lookups)
CREATE INDEX IF NOT EXISTS escalation_feedback_team_level_idx
    ON escalation_feedback (human_team, human_level, was_overridden);


-- ─────────────────────────────────────────────────────────────
-- MIGRATE: Add escalation columns to the existing tickets table
-- These columns store the AI escalation result per ticket.
-- Run each line individually if any column already exists.
-- ─────────────────────────────────────────────────────────────

ALTER TABLE tickets
    ADD COLUMN IF NOT EXISTS esc_team        TEXT,       -- e.g. "Network Ops"
    ADD COLUMN IF NOT EXISTS esc_level       TEXT,       -- L1 / L2 / L3
    ADD COLUMN IF NOT EXISTS esc_confidence  TEXT,       -- HIGH / MEDIUM / LOW
    ADD COLUMN IF NOT EXISTS esc_action      TEXT,       -- AUTO_ROUTED / HITL_PENDING / QUARANTINE / HUMAN_FINALISED
    ADD COLUMN IF NOT EXISTS esc_rationale   TEXT,       -- AI or human explanation
    ADD COLUMN IF NOT EXISTS escalated_at    TIMESTAMPTZ;


-- ─────────────────────────────────────────────────────────────
-- FUNCTION: Count human confirmations for self-learning
-- Returns how many times humans have confirmed (not overridden)
-- a given team+level. Used by service.py to decide if AI should
-- get a confidence boost for this routing path.
-- ─────────────────────────────────────────────────────────────

CREATE OR REPLACE FUNCTION get_escalation_feedback_count(
    p_team   TEXT,
    p_level  TEXT
)
RETURNS INTEGER
LANGUAGE sql STABLE AS $$
    SELECT COUNT(*)::INTEGER
    FROM escalation_feedback
    WHERE human_team   = p_team
      AND human_level  = p_level
      AND was_overridden = FALSE;
$$;
